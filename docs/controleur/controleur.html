263def limite_bord_et_diago(joueur, dx, dy):
264    """Empêche le joueur de dépasser les bords de l'écran et change la vitesse en fonction de diagonale ou pas
265    Renvoie le déplacement et la bonne vitesse"""
266    # Si déplacement diagonal : la vitesse est adaptée
267    if dx != 0 and dy != 0:
268        facteur = joueur.vitesse / math.sqrt(2)
269    else:
270        facteur = joueur.vitesse
271
272    # Si le déplacement va dépasser le cadre, on ne déplace que vers la limite du bord
273    bord_droit = joueur.x + joueur.taille
274    bord_gauche = joueur.x - joueur.taille
275    bord_haut = joueur.y - joueur.taille
276    bord_bas = joueur.y + joueur.taille
277    if bord_droit + facteur >= configuration.largeur and dx == 1:
278        joueur.tp_bord(configuration.largeur - bord_droit, "horizontal")
279        dx = 0
280    if bord_gauche - facteur <= 0 and dx == -1:
281        joueur.tp_bord(0 - bord_gauche, "horizontal")
282        dx = 0
283    if bord_haut - facteur <= 0 and dy == -1:
284        joueur.tp_bord(0 - bord_haut, "vertical")
285        dy = 0
286    if bord_bas + facteur >= configuration.hauteur and dy == 1:
287        joueur.tp_bord(configuration.hauteur - bord_bas, "vertical")
288        dy = 0
289    return dx, dy, facteur

def limite_bord_et_diago(joueur, dx, dy): 
264def limite_bord_et_diago(joueur, dx, dy):
265    """Empêche le joueur de dépasser les bords de l'écran et change la vitesse en fonction de diagonale ou pas
266    Renvoie le déplacement et la bonne vitesse"""
267    # Si déplacement diagonal : la vitesse est adaptée
268    if dx != 0 and dy != 0:
269        facteur = joueur.vitesse / math.sqrt(2)
270    else:
271        facteur = joueur.vitesse
272
273    # Si le déplacement va dépasser le cadre, on ne déplace que vers la limite du bord
274    bord_droit = joueur.x + joueur.taille
275    bord_gauche = joueur.x - joueur.taille
276    bord_haut = joueur.y - joueur.taille
277    bord_bas = joueur.y + joueur.taille
278    if bord_droit + facteur >= configuration.largeur and dx == 1:
279        joueur.tp_bord(configuration.largeur - bord_droit, "horizontal")
280        dx = 0
281    if bord_gauche - facteur <= 0 and dx == -1:
282        joueur.tp_bord(0 - bord_gauche, "horizontal")
283        dx = 0
284    if bord_haut - facteur <= 0 and dy == -1:
285        joueur.tp_bord(0 - bord_haut, "vertical")
286        dy = 0
287    if bord_bas + facteur >= configuration.hauteur and dy == 1:
288        joueur.tp_bord(configuration.hauteur - bord_bas, "vertical")
289        dy = 0
290    return dx, dy, facteur
